  ```typescript
  export class CellPropertyDefinition {
    readonly name: string;
    readonly type: CellPropertyType;
    readonly channels: number;
    readonly defaultValue: number[];  // Always stored as array internally
    readonly role: PropertyRole;
    readonly isComputed: boolean;

    constructor(config: CellPropertyConfig) { ... }

    /** Get the default value for this property type */
    getDefault(): number | number[] { ... }

    /** Get the channel count for this property */
    getChannels(): number { ... }

    /** Validate a value against this property's type */
    validateValue(value: number | number[]): boolean { ... }

    /** Read a value from a Float32Array at the given offset */
    readFromBuffer(buffer: Float32Array, cellIndex: number, propertyOffset: number): number | number[] { ... }

    /** Write a value to a Float32Array at the given offset */
    writeToBuffer(buffer: Float32Array, cellIndex: number, propertyOffset: number, value: number | number[]): void { ... }
  }
  ```

  Key behaviors:
  - Bool: default is 0.0, read returns 0 or 1 (Math.round), write clamps to 0.0/1.0
  - Int: default is 0, read returns Math.round(float), write stores as float
  - Float: stored directly
  - Vec2/vec3/vec4: default is array of zeros matching channel count
  - Single number defaults for vector types expand to fill all channels
  - Role defaults to 'input_output' if not specified
  - `isComputed` is true when config has a `compute` string

  CRITICAL: No browser API imports.

  <action>
  Create `src/engine/cell/ComputedFunction.ts`:

  ```typescript
  export class ComputedFunction {
    readonly propertyName: string;
    private fn: ComputeFn;
    private source: string;

    constructor(propertyName: string, functionBody: string) {
      this.propertyName = propertyName;
      this.source = functionBody;
      this.fn = this.compile(functionBody);
    }

    /** Compile a function body string into a callable function */
    private compile(body: string): ComputeFn {
      // Uses new Function() to compile the body string
      // The function receives a single 'ctx' parameter of type ComputeContext
      return new Function('ctx', body) as ComputeFn;
    }

    /** Execute the computed function with the given context */
    evaluate(ctx: ComputeContext): number | number[] {
      return this.fn(ctx);
    }

    /** Get the original source for debugging */
    getSource(): string {
      return this.source;
    }
  }
  ```

  Example usage in tests:
  ```typescript
  const fn = new ComputedFunction('alive', 'return ctx.cell.alive ? 1 : 0;');
  const result = fn.evaluate({ cell: { alive: 1 }, neighbors: [], grid: {...}, params: {} });
  ```

  Error handling: If the function body has syntax errors, the constructor should throw with a clear message indicating which property's compute function failed.
  </action>
